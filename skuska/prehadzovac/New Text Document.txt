    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        jTextField2.setText("");
        String[] Slova = jTextField1.getText().split(" ");

        for (int j = 0; j < Slova.length; j++) {
            String slovo = Slova[j];

            StringBuilder sb = new StringBuilder(slovo.length());

            ArrayList pismena = new ArrayList();
            for (int i = 1; i < slovo.length() - 1; i++) {
                pismena.add(slovo.charAt(i));
            }

            Random rnd = new Random();

            sb.append(slovo.charAt(0));

            int size = pismena.size();
            for (int i = 0; i < size; i++) {
                int cislo = rnd.nextInt(pismena.size());
                char znak = (char) pismena.get(cislo);
                pismena.remove(cislo);
                sb.append(znak);
            }

            sb.append(slovo.charAt(slovo.length() - 1));
            jTextField2.setText(jTextField2.getText() + " " + sb.toString());
        }
    }
